fun Content(
    viewModel: EditorViewModel = viewModel(),
    state: EditorUIState,

        onRetry = {
            viewModel.refresh()
        }

                LoadingDialog()

                scope.launch {
                    viewModel.showErrorSnackbar(state.message ?: "Unexpected Error")

            is EditorUIState.Success -> {

                    value = viewModel.editorText.value,
                    onValueChange = { viewModel.editorText.value = it }
                )
            }
        }
    }
}
